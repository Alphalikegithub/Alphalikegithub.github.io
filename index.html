import pandas as pd

# 第一步：读取并处理Excel文件
input_excel_file = 'srcfile/正偏差.xlsx'
output_excel_file = 'out/正偏差管道累加弹道.xlsx'
output_txt_file = 'out/正偏差管道累加弹道.txt'

# 读取Excel文件
df = pd.read_excel(input_excel_file)

# 进行列数据的相加
df['x+dX'] = df['x'] + df['dX']
df['y+dY'] = df['y'] + df['dY']
df['z+dZ'] = df['z'] + df['dZ']
df['Vx+dVx'] = df['Vx'] + df['dVx']
df['Vy+dVy'] = df['Vy'] + df['dVy']
df['Vz+dVz'] = df['Vz'] + df['dVz']

# 选择所需的列进行保存
result_df = df[['t1', 'x+dX', 'y+dY', 'z+dZ', 'Vx+dVx', 'Vy+dVy', 'Vz+dVz']]

# 导出为新的Excel文件
result_df.to_excel(output_excel_file, index=False)

print(f"正偏差累加数据已成功导出到 {output_excel_file}")

# 第二步：读取并导出为TXT文件
# 读取Excel文件
df = pd.read_excel(output_excel_file)

# 将所有行数据写入TXT文件
df.to_csv(output_txt_file, sep='\t', index=False)

print(f"所有行数据已成功导出到 {output_txt_file}")

版本二（使用列的索引）

你可以通过列的索引来指定要进行操作的列，而不是使用列名。这可以通过Pandas的.iloc属性来实现。以下是一个示例代码，它将读取Excel文件，进行指定列的相加操作，并导出为新的Excel文件和TXT文件。

示例代码 

import pandas as pd

# 第一步：读取并处理Excel文件
input_excel_file = 'srcfile/正偏差.xlsx'
output_excel_file = 'out/简单正偏差累加3.xlsx'
output_txt_file = 'out/简单正偏差累加3.txt'

# 读取Excel文件
df = pd.read_excel(input_excel_file)

# 进行列数据的相加
df['x'] = df.iloc[:, 1] + df.iloc[:, 9]
df['y'] = df.iloc[:, 2] + df.iloc[:, 10]
df['z'] = df.iloc[:, 3] + df.iloc[:, 11]
df['Vx'] = df.iloc[:, 4] + df.iloc[:, 13]
df['Vy'] = df.iloc[:, 5] + df.iloc[:, 14]
df['Vz'] = df.iloc[:, 6] + df.iloc[:, 15]

# 选择所需的列进行保存
result_df = df[['t1', 'x', 'y', 'z', 'Vx', 'Vy', 'Vz']]

# 导出为新的Excel文件
result_df.to_excel(output_excel_file, index=False)

print(f"正偏差累加数据已成功导出到 {output_excel_file}")

# 第二步：读取并导出为TXT文件
# 读取Excel文件
df = pd.read_excel(output_excel_file)

# 将所有行数据写入TXT文件
df.to_csv(output_txt_file, sep='\t', index=False)

print(f"所有行数据已成功导出到 {output_txt_file}")

说明

  • df.iloc[:, 0] 表示第1列，df.iloc[:, 3] 表示第4列，依次类推。
  • 使用 iloc 可以根据列的索引进行操作，而不是列名。
  • 最终结果保存在新的Excel文件和TXT文件中。

请确保你的列数和索引匹配，并根据需要调整代码。如果列数更多，你可以继续扩展相应的操作。

def process_deviation(input_file, output_excel_file, output_txt_file, deviation_columns, custom_headers):
    # 读取Excel文件
    df = pd.read_excel(input_file)
    
    # 进行列数据的处理
    for col, deviation_col in deviation_columns.items():
        df[f'{col}+{deviation_col}'] = df[col] + df[deviation_col]
    
    # 选择所需的列进行保存
    result_df = df[['t1'] + [f'{col}+{deviation_col}' for col, deviation_col in deviation_columns.items()]]
    
    # 修改表头
    result_df.columns = ['t1'] + custom_headers
    
    # 导出为新的Excel文件
    result_df.to_excel(output_excel_file, index=False)
    print(f"数据已成功导出到 {output_excel_file}")
    
    # 将所有行数据写入TXT文件
    result_df.to_csv(output_txt_file, sep='\t', index=False)
    print(f"数据已成功导出到 {output_txt_file}")
